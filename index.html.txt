function render() {
    let html = "";
    tasks.forEach((t) => {
        const status = taskStatus[t.id];
        const isDone = status === 'done';
        const isSkip = status === 'skip';
        html += `
            <div class="card ${isDone ? 'completed' : (isSkip ? 'skip' : '')}">
                <div class="task-name">${t.name}</div>
                <div class="status-msg done">✅ 已完成！</div>
                <div class="status-msg no-need">☕ 今日免做</div>
                <div class="btn-group">
                    <button class="btn-prompt" ${status ? 'disabled' : ''} onclick="update(${t.id}, ${t.p}, 'done')">提醒 +${t.p}</button>
                    <button class="${t.type==='study'?'btn-study':'btn-active'}" ${status ? 'disabled' : ''} onclick="update(${t.id}, ${t.a}, 'done')">主動 +${t.a}</button>
                    <button class="btn-skip" ${status ? 'disabled' : ''} onclick="update(${t.id}, 0, 'skip')">免做</button>
                </div>
            </div>
        `;
    });
    document.getElementById("list").innerHTML = html;
    document.getElementById("score").innerText = score;
}

function update(id, val, mode) {
    if (taskStatus[id]) return;
    score += val;
    taskStatus[id] = mode;
    localStorage.setItem('my_score', score);
    localStorage.setItem('my_task_status', JSON.stringify(taskStatus));
    render();
}

function resetScore() {
    if(!confirm("確定要重置嗎？")) return;
    score = 0; taskStatus = {}; localStorage.clear(); render();
}

window.onload = function() {
    score = parseInt(localStorage.getItem('my_score')) || 0;
    render();
}
